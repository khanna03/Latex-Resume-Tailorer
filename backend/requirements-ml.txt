# ==============================================================================
# LaTeX Resume Tailorer — ML Training Dependencies (Phase 5/6)
# ==============================================================================
# This file specifies all Python libraries required for fine-tuning local LLMs.
# Run 'pip install -r requirements-ml.txt' in a dedicated training environment.
#
# IMPORTANT: These dependencies are intentionally separate from the main
# requirements.txt to avoid bloating the web server's Docker image.
# PyTorch alone is ~2GB, and you only need these on a GPU machine for training.
# ==============================================================================

# ---- PyTorch (Deep Learning Framework) ----
# The foundation for all neural network computations. Provides GPU acceleration
# via CUDA, automatic differentiation (autograd), and the tensor operations
# that power transformer models.
torch>=2.0.0

# ---- HuggingFace Transformers ----
# The main library for loading pretrained LLMs (LLaMA, Mistral, Qwen, etc.)
# and their tokenizers. Provides AutoModelForCausalLM, AutoTokenizer, and
# TrainingArguments which configure the entire training loop.
transformers>=4.40.0

# ---- PEFT (Parameter-Efficient Fine-Tuning) ----
# Implements LoRA (Low-Rank Adaptation) and other adapter techniques.
# Instead of updating all ~8 billion model parameters during training,
# LoRA injects small trainable matrices into attention layers, reducing
# trainable parameters from 8B to ~13M (0.16%) and VRAM from 32GB to ~6GB.
peft>=0.10.0

# ---- TRL (Transformer Reinforcement Learning) ----
# Provides SFTTrainer (Supervised Fine-Tuning Trainer) which handles
# the training loop, dataset formatting, and integration with PEFT adapters.
# Also supports RLHF and DPO for more advanced alignment training.
trl>=0.8.6

# ---- HuggingFace Datasets ----
# Memory-efficient dataset library backed by Apache Arrow.
# Supports lazy loading, memory mapping, and streaming for large datasets.
datasets>=2.19.0

# ---- Accelerate ----
# HuggingFace's distributed training library. Handles device placement
# (CPU/GPU/multi-GPU), mixed precision (FP16/BF16), and gradient accumulation
# automatically. Required by TrainingArguments and SFTTrainer.
accelerate>=0.30.1

# ---- BitsAndBytes ----
# Provides 4-bit and 8-bit quantization for PyTorch models (QLoRA).
# Compresses model weights from 16-bit to 4-bit, reducing VRAM usage by ~4x.
# Also provides paged optimizers (paged_adamw_8bit) that use less memory
# by offloading optimizer states to CPU RAM when GPU VRAM is full.
bitsandbytes>=0.43.1
